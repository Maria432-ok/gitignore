import pandas as pd

# --- PASO 1: CARGAR EL ARCHIVO CSV ---
try:
    df = pd.read_csv('vehicles_us.csv')
    
    print("✅ Archivo cargado exitosamente.")
    print("-" * 30)

    # --- PASO 2: REVISIÓN INICIAL DE DATOS ---
    
    # 2.1 Muestra la información general (tipos de datos y valores no nulos)
